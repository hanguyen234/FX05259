image = new SimpleImage("flowers.jpg");

for (pixel: image) {
  if(pixel.getY() > image.getHeight()/2){
    pixel.setRed(255 - pixel.getRed());
    pixel.setGreen(255 - pixel.getGreen());
    pixel.setBlue(255 - pixel.getBlue());
  }
}

print(image);
